```js
async function requireAuth() {
  const { data: { session } } = await db.auth.getSession();
  if (!session) {
    window.location.href = '/login.html';
    return null;
  }
  const emailEl = document.getElementById('user-email');
  if (emailEl) emailEl.textContent = session.user.email;
  return session;
}

async function signOut() {
  await db.auth.signOut();
  window.location.href = '/login.html';
}
```

```js
async function loadActionStrip() {
  const now = new Date().toISOString();
  const cutoff14 = new Date(Date.now() - 14 * 24 * 60 * 60 * 1000).toISOString();

  const [{ data: clients }, { data: partners }] = await Promise.all([
    db.from('clients').select('lifecycle_stage,activated_at,review_asked_at,review_received_at,referral_asked_at,created_at'),
    db.from('partners').select('activation_status')
  ]);

  const signals = {
    notActivated: clients.filter(c => c.lifecycle_stage === 'onboarding' && !c.activated_at).length,
    overdue: clients.filter(c => c.lifecycle_stage === 'onboarding' && !c.activated_at && c.created_at < cutoff14).length,
    noReview: clients.filter(c => c.activated_at && !c.review_asked_at).length,
    partnerPending: partners.filter(p => p.activation_status === 'pending').length,
    referralOppty: clients.filter(c => c.review_received_at && !c.referral_asked_at).length
  };

  const strip = document.getElementById('action-strip');
  strip.innerHTML = `
    <span class="action-strip-label">Action needed</span>
    ${signal(signals.notActivated, 'Not activated')}
    ${signal(signals.overdue, '14+ days overdue', 'warn')}
    ${signal(signals.noReview, 'No review ask')}
    ${signal(signals.partnerPending, 'Partners pending')}
    ${signal(signals.referralOppty, 'Referral opportunity')}
  `;
}

function signal(count, label, type = '') {
  const cls = count === 0 ? 'zero' : (type === 'warn' ? 'warn' : '');
  return `<span class="action-signal"><span class="action-signal-count${cls ? ' action-signal-count--' + cls : ''}">${count}</span>${label}</span>`;
}
```

```js
  const now = Date.now();
  const cutoff14 = now - 14 * 24 * 60 * 60 * 1000;

  document.getElementById('rail-list').innerHTML = sorted.length === 0

    : sorted.map(c => {
        const overdue = c.lifecycle_stage === 'onboarding' && !c.activated_at && new Date(c.created_at) < cutoff14;
        return `<div class="rail-item${c.id === selectedClientId ? ' active' : ''}" data-id="${c.id}">
```

```js
async function renderCostsTab(client) {
  const { data: projects } = await db.from('projects')
    .select('id,name,retainer_rate,start_date,status').eq('client_id', client.id);

  const projectIds = projects.map(p => p.id);
  let costs = [];
  if (projectIds.length > 0) {
    const { data } = await db.from('project_costs').select('*').in('project_id', projectIds);
    costs = data || [];
  }

  const today = new Date();
  let totalRevenue = 0;
  let totalCosts = 0;

  const projectRows = projects.map(p => {
    const start = p.start_date ? new Date(p.start_date) : today;
    const daysDiff = Math.max(0, Math.floor((today - start) / (1000 * 60 * 60 * 24)));
    const monthsActive = Math.max(1, Math.floor(daysDiff / 30));
    const revenue = (p.retainer_rate || 0) * monthsActive;
    const pCosts = costs.filter(c => c.project_id === p.id).reduce((s, c) => s + (c.amount || 0), 0);
    const net = revenue - pCosts;
    const margin = revenue > 0 ? Math.round(net / revenue * 100) : 0;
    totalRevenue += revenue;
    totalCosts += pCosts;
    return { p, monthsActive, revenue, pCosts, net, margin };
```

```js
async function selectPartner(id) {
  selectedPartnerId = id;
  renderRail(allPartners);
  const partner = allPartners.find(p => p.id === id);
  if (!partner) return;
  renderDetailShell(partner);
  attachTabListeners(partner);
  await loadTab('overview', partner);
}

function renderDetailShell(partner) {
  const tabs = ['Overview'];
  if (partner.type === 'agent') tabs.push('Clients Referred');
  if (['supplier','employee'].includes(partner.type)) tabs.push('Projects');
  tabs.push('Activation');

  document.getElementById('detail-pane').innerHTML = `
    <div class="detail-header">
      <div class="detail-header-top">
        <div>
          <div class="detail-name">${partner.name}</div>

        </div>
        <span class="badge badge--${partner.type}">${partner.type}</span>
      </div>
      <div class="tabs">
        ${tabs.map((t, i) => `<button class="tab${i===0?' active':''}" data-tab="${t.toLowerCase().replace(' ','_')}">${t}</button>`).join('')}
      </div>
    </div>
    <div class="detail-body" id="tab-content">Loading…</div>
  `;
}

function attachTabListeners(partner) {
  document.querySelectorAll('.tab').forEach(tab => {
    tab.addEventListener('click', async () => {
      document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
      tab.classList.add('active');
      document.getElementById('tab-content').innerHTML = 'Loading…';
      await loadTab(tab.dataset.tab, partner);
    });
  });
}

async function loadTab(tabName, partner) {
  switch (tabName) {
    case 'overview':         renderPartnerOverview(partner); break;
    case 'clients_referred': await renderClientsReferred(partner); break;
    case 'projects':         await renderPartnerProjects(partner); break;
    case 'activation':       await renderPartnerActivation(partner); break;
  }
}

function showAddPartnerModal() {
  const name = prompt('Partner name:');
  if (!name?.trim()) return;
  const type = prompt('Type (agent/supplier/employee):');
  if (!['agent','supplier','employee'].includes(type)) { alert('Invalid type'); return; }
  db.from('partners').insert({ name: name.trim(), type }).select().single().then(({ data, error }) => {
    if (error) { alert('Error: ' + error.message); return; }
    allPartners.push(data);
    renderRail(allPartners);
    selectPartner(data.id);
  });
}
```
